\section*{\setsectioncolor{BrightPurple} PART A — Qüestions Test (4 punts)}

\textit{Penalització: Encert +0.4 | Error -0.13 | En blanc 0}

\begin{enumerate}[label=\textbf{\arabic*.}]
    \item Quin és el resultat de l'operació $-5 + (-3) \cdot (-2)$?
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item 16 \item 1 \item -11 \item -16
    \end{enumerate}
    \end{multicols}

    \item Si la temperatura a les 6:00 era de $-4^\circ\text{C}$ i a les 12:00 és de $7^\circ\text{C}$, quants graus ha pujat?
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item $3^\circ\text{C}$ \item $-11^\circ\text{C}$ \item $11^\circ\text{C}$ \item $4^\circ\text{C}$
    \end{enumerate}
    \end{multicols}

    \item Quina d'aquestes fraccions és equivalent a $\frac{12}{18}$?
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item $\frac{2}{3}$ \item $\frac{6}{8}$ \item $\frac{3}{2}$ \item $\frac{4}{9}$
    \end{enumerate}
    \end{multicols}

    \item El resultat de la operació $\frac{3}{4} + \frac{1}{2}$ és:
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item $\frac{4}{6}$ \item $\frac{4}{8}$ \item $\frac{5}{4}$ \item $\frac{1}{4}$
    \end{enumerate}
    \end{multicols}

    \item Calcula la fracció irreducible de $\frac{45}{75}$:
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item $\frac{9}{15}$ \item $\frac{3}{5}$ \item $\frac{5}{3}$ \item $\frac{15}{25}$
    \end{enumerate}
    \end{multicols}

    \item Quin és el resultat de $\frac{2}{3} \cdot \frac{5}{4}$?
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item $\frac{7}{7}$ \item $\frac{10}{12}$ \item $\frac{5}{6}$ \item $\frac{8}{15}$
    \end{enumerate}
    \end{multicols}

    \item Si dividim $\frac{3}{5} \div \frac{2}{7}$, el resultat és:
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item $\frac{6}{35}$ \item $\frac{5}{12}$ \item $\frac{21}{10}$ \item $\frac{10}{21}$
    \end{enumerate}
    \end{multicols}

    \item En una classe de 30 alumnes, els $\frac{3}{5}$ són noies. Quants nois hi ha?
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item 18 \item 12 \item 15 \item 20
    \end{enumerate}
    \end{multicols}

    \item Quina és l'opció correcta per a $-2^2$?
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item 4 \item -4 \item -2 \item 1
    \end{enumerate}
    \end{multicols}

    \item Quina fracció representa la part pintada d'un pastís si n'hem menjat $\frac{1}{4}$ i després la meitat del que quedava?
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item $\frac{1}{2}$ \item $\frac{3}{4}$ \item $\frac{3}{8}$ \item $\frac{5}{8}$
    \end{enumerate}
    \end{multicols}
\end{enumerate}

\noindent{\large\bfseries Problema 1 (2 punts): L'ascensor del centre comercial} \\
Un ascensor es troba a la planta $-2$. Puja 5 plantes, després en baixa 3, torna a pujar 7 i finalment en baixa 8. \\
\noindent\textbf{a) (1 punt)} En quina planta es troba finalment? \\
\vspace{4cm}
\noindent\textbf{b) (1 punt)} Si cada planta fa 3 metres d'alçada, quants metres ha recorregut en total (pujant i baixant)? \\
\vspace{4cm}
\begin{tcolorbox}[colback=blue!5, colframe=blue!30, title=Comprovació de seguretat]
    La planta final és una planta per sobre de la planta baixa (0).
\end{tcolorbox}

\noindent{\large\bfseries Problema 3 (2 punts): El dipòsit de benzina} \\
El dipòsit de benzina d'un cotxe té una capacitat de 60 litres. El viatge d'anada ha consumit $\frac{2}{5}$ del dipòsit i el de tornada $\frac{1}{3}$ del dipòsit inicial. \\
\noindent\textbf{a) (1 punt)} Quina fracció del dipòsit queda al final del viatge? \\
\vspace{6cm}
\noindent\textbf{b) (1 punt)} Quants litres de benzina s'han consumit en total? \\
\vspace{6cm}
\begin{tcolorbox}[colback=blue!5, colframe=blue!30, title=Comprovació de seguretat]
    Queden menys de 20 litres al dipòsit.
