\parttitle{Part A — Qüestions (4 punts)}

\begin{enumerate}[label=\textbf{\arabic*.}, itemsep=0.8em]

    \item Quin és el resultat de $3 + (-7)$?
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $10$ \item $-4$ \item $4$ \item $-10$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: B)} \textit{$3+(-7)=3-7=-4$.}}

    \item La temperatura era de $-3\,^{\circ}\text{C}$ i ha baixat $5\,^{\circ}\text{C}$. A quina temperatura estem?
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $2\,^{\circ}\text{C}$ \item $-2\,^{\circ}\text{C}$ \item $-8\,^{\circ}\text{C}$ \item $8\,^{\circ}\text{C}$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: C)} \textit{$-3-5=-8\,^{\circ}\text{C}$.}}

    \item Una cinta es talla en 6 trossos iguals. En tenim 2. Quina fracció tenim?
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $\dfrac{6}{2}$ \item $\dfrac{1}{3}$ \item $\dfrac{2}{6}$ \item $\dfrac{3}{1}$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: B o C)} \textit{$\frac{2}{6}=\frac{1}{3}$. Ambdues són correctes; la forma irreducible és B ($\frac{1}{3}$).}}

    \item Calcula $\dfrac{2}{5} + \dfrac{1}{5}$:
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $\dfrac{3}{10}$ \item $\dfrac{3}{5}$ \item $\dfrac{2}{5}$ \item $\dfrac{1}{5}$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: B)} \textit{Mateix denominador: $\frac{2+1}{5}=\frac{3}{5}$.}}

\end{enumerate}

\parttitle{Part B — Problemes (6 punts)}

\noindent{\large\bfseries Problema 1 (3 punts): El termòmetre}

\noindent Al matí el termòmetre marca $-5\,^{\circ}\text{C}$. Al migdia ha pujat $9\,^{\circ}\text{C}$. A la nit baixa $4\,^{\circ}\text{C}$.

\noindent\textbf{a)} Quina temperatura fa al migdia?

\textcolor{red}{$-5 + 9 = \mathbf{4\,^{\circ}\text{C}}$}
\vspace{0.5cm}

\noindent\textbf{b)} Quina temperatura fa a la nit?

\textcolor{red}{$4 - 4 = \mathbf{0\,^{\circ}\text{C}}$}
\vspace{1cm}

\noindent{\large\bfseries Problema 2 (3 punts): El pastís d'aniversari}

\noindent Un pastís es talla en 8 porcions iguals. En mengem 3 porcions.

\noindent\textbf{a)} Quina fracció del pastís hem menjat? Escriu-la de forma irreducible.

\textcolor{red}{$\dfrac{3}{8}$ \quad MCD(3,8)=1 $\Rightarrow$ ja és irreducible.}
\vspace{0.5cm}

\noindent\textbf{b)} Quantes porcions queden?

\textcolor{red}{$8 - 3 = \mathbf{5}$ porcions.}
