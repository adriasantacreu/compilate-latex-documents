\begin{enumerate}[label=\textbf{\arabic*.}, itemsep=0.8em]

    \item Calcula: $-5 + 6 \cdot 2$
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $7$ \item $2$ \item $-11$ \item $1$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: A)} \textit{Jerarquia: $6\cdot2=12$, llavors $-5+12=7$.}}

    \item A la nevera estem a $-4\,^{\circ}\text{C}$ i la temperatura puja $11\,^{\circ}\text{C}$. A quina temperatura estem?
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $7\,^{\circ}\text{C}$ \item $15\,^{\circ}\text{C}$ \item $-7\,^{\circ}\text{C}$ \item $0\,^{\circ}\text{C}$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: A)} \textit{$-4+11=7\,^{\circ}\text{C}$.}}

    \item Quina fracció és equivalent a $\dfrac{1}{2}$?
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $\dfrac{2}{4}$ \item $\dfrac{2}{3}$ \item $\dfrac{1}{4}$ \item $\dfrac{3}{2}$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: A)} \textit{$\frac{2}{4}=\frac{1}{2}$ (dividint per 2).}}

    \item Calcula $\dfrac{3}{4} + \dfrac{1}{4}$:
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $\dfrac{4}{8}$ \item $\dfrac{4}{4} = 1$ \item $\dfrac{2}{4}$ \item $\dfrac{3}{16}$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: B)} \textit{$\frac{3+1}{4}=\frac{4}{4}=1$.}}

\end{enumerate}

\parttitle{Part B — Problemes (6 punts)}

\noindent{\large\bfseries Problema 1 (3 punts): L'ascensor}

\noindent Estem a la planta $-2$. L'ascensor puja 5 plantes i desprès en baixa 8.

\noindent\textbf{a)} A quina planta arribem al final?

\textcolor{red}{
\begin{align*}
-2 + 5 &= +3 \quad \text{(puja 5)} \\
3 - 8 &= \mathbf{-5} \quad \text{(baixa 8) → planta final: \textbf{-5}}
\end{align*}
}
\vspace{0.5cm}

\noindent{\large\bfseries Problema 2 (3 punts): El dipòsit}

\noindent Tenim un dipòsit de 60 litres. N'hem consumit $\dfrac{1}{4}$ part.

\noindent\textbf{a)} Quants litres hem gastat?

\textcolor{red}{$\dfrac{1}{4} \times 60 = \dfrac{60}{4} = \mathbf{15 \text{ litres}}$}
\vspace{0.5cm}

\noindent\textbf{b)} Quants litres queden al dipòsit?

\textcolor{red}{$60 - 15 = \mathbf{45 \text{ litres}}$}
