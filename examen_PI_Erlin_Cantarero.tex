\section*{\setsectioncolor{BrightPurple} PART A — Qüestions (4 punts)}

\begin{enumerate}[label=\textbf{\arabic*.}]
    \item Quin és el resultat de $-5 + 6$?
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item 1 \item -11 \item 11 \item -1
    \end{enumerate}
    \end{multicols}

    \item Si a la nevera estem a $-4^\circ\text{C}$ i pugem $7^\circ\text{C}$, a quina temperatura estarem?
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item $3^\circ\text{C}$ \item $11^\circ\text{C}$ \item $-11^\circ\text{C}$ \item $0^\circ\text{C}$
    \end{enumerate}
    \end{multicols}

    \item Quina fracció és la meitat d'un pastís?
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item $\frac{1}{4}$ \item $\frac{1}{2}$ \item $\frac{2}{1}$ \item $\frac{1}{3}$
    \end{enumerate}
    \end{multicols}

    \item Calcula $\frac{1}{4} + \frac{2}{4}$:
    \begin{multicols}{4}
    \begin{enumerate}[label=\Alph*)]
        \item $\frac{3}{8}$ \item $\frac{3}{4}$ \item $\frac{2}{4}$ \item $\frac{4}{4}$
    \end{enumerate}
    \end{multicols}
\end{enumerate}
