\parttitle{Part A — Qüestions (4 punts)}

\begin{enumerate}[label=\textbf{\arabic*.}, itemsep=0.8em]

    \item Quin és el resultat de $-5 + 6$?
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $1$ \item $-11$ \item $11$ \item $-1$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: A)} \textit{$-5+6=1$.}}

    \item La temperatura era de $-4\,^{\circ}\text{C}$ i ha pujat $7\,^{\circ}\text{C}$. A quina temperatura estem?
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $3\,^{\circ}\text{C}$ \item $11\,^{\circ}\text{C}$ \item $-11\,^{\circ}\text{C}$ \item $0\,^{\circ}\text{C}$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: A)} \textit{$-4+7=3\,^{\circ}\text{C}$.}}

    \item Quina fracció és la meitat d'un pastís?
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $\dfrac{1}{4}$ \item $\dfrac{1}{2}$ \item $\dfrac{2}{1}$ \item $\dfrac{1}{3}$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: B)} \textit{La meitat és dividir en 2 parts iguals: $\frac{1}{2}$.}}

    \item Calcula $\dfrac{1}{4} + \dfrac{2}{4}$:
    \begin{multicols}{4}
    \begin{enumerate}[label=\textbf{\Alph*)}, itemsep=0.3em]
        \item $\dfrac{3}{8}$ \item $\dfrac{3}{4}$ \item $\dfrac{2}{4}$ \item $\dfrac{4}{4}$
    \end{enumerate}
    \end{multicols}
    \textcolor{red}{\textbf{Resp: B)} \textit{Mateix denominador: $\frac{1+2}{4}=\frac{3}{4}$.}}

\end{enumerate}

\parttitle{Part B — Problemes (6 punts)}

\noindent{\large\bfseries Problema 1 (3 punts): L'ascensor}

\noindent Estem a la planta $-2$. L'ascensor puja 5 plantes.

\noindent\textbf{a)} A quina planta arribem?

\textcolor{red}{$-2 + 5 = \mathbf{+3}$ \quad (planta 3)}
\vspace{0.5cm}

\noindent\textbf{b)} Des d'aquesta planta, baixem 3 plantes més. On som ara?

\textcolor{red}{$3 - 3 = \mathbf{0}$ \quad (planta baixa, planta 0)}
\vspace{1cm}

\noindent{\large\bfseries Problema 2 (3 punts): Berenar amb amics}

\noindent Tenim una pizza tallada en 8 trossos. En mengem 3 trossos.

\noindent\textbf{a)} Quina fracció de la pizza hem menjat?

\textcolor{red}{Hem menjat 3 trossos de 8: $\dfrac{3}{8}$ de la pizza. (MCD(3,8)=1, ja és irreducible.)}
\vspace{0.5cm}

\noindent\textbf{b)} Quants trossos queden a la caixa?

\textcolor{red}{$8 - 3 = \mathbf{5}$ trossos.}
